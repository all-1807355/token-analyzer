    function swapandliquify(uint256 contracttokenbalance) private locktheswap {
        // 2% is sent to liquidity and 6% is sent to house
        // this means that liquidity is 1/4 of the total balance
        uint256 liquiditytokenbalance = contracttokenbalance.div(4);
        uint256 housetokenbalance = contracttokenbalance.sub(liquiditytokenbalance);

        // split the token balance to be liquified into halves, this is so we can deposit the same amount
        // of bnb and tokens into the lp
        uint256 tokenbalancetoliquifyasbnb = liquiditytokenbalance.div(2);
        uint256 tokenbalancetoliquify = liquiditytokenbalance.sub(tokenbalancetoliquifyasbnb);

        // capture the contract's current bnb balance.
        // this is so that we can capture exactly the amount of bnb that the
        // swap creates, and not make the liquidity event include any bnb that
        // has been manually sent to the contract
        uint256 initialbalance = address(this).balance;


        // 1/8 = tokens to liquify
        // 1/8 = bnb to liquify
        // 6/8 = bnb to send to house
        uint256 tokenstoswaptobnb = tokenbalancetoliquifyasbnb.add(housetokenbalance);

        // swap tokens for bnb
        swaptokensforeth(tokenstoswaptobnb);

        // total bnb that has been swapped
        uint256 bnbswapped = address(this).balance.sub(initialbalance);

        // bnb to liquify is 1/8 of the total token balance or 1/7 of the bnb that has already been liquified
        uint256 bnbtoliquify = bnbswapped.div(7);

        // add liquidity to pancake swap
        addliquidity(tokenbalancetoliquify, bnbtoliquify);

        emit swapandliquify(tokenbalancetoliquifyasbnb, bnbtoliquify, tokenbalancetoliquify);

        // the remaining bnb balance is to be donated to the house wallet
        uint256 housebnbtodonate = bnbswapped.sub(bnbtoliquify);

        // transfer the bnb to the house wallet
        (bool success, ) = _housewallet.call.value(housebnbtodonate)("");
        require(success, "unable to send to sc");
    }
